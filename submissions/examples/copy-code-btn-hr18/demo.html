EaseMotion-css · component

Copy Code button

A drop-in copy button for documentation code blocks — installation commands, CDN links, and import statements can all be copied with one click instead of manual selection.

Install via npm 
npm install easemotion-css
Or via CDN 
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/easemotion-css/dist/easemotion.min.css">
Import in your entry file 
// app/layout.js
import "easemotion-css/min";

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <body>{children}</body>
    </html>
  );
}
