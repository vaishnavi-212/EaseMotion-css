Implementation Quick Start

CSS Usage
style.css & demo.html 
/* 1. Optical Blur-Entrance Keyframe Animation */
@keyframes emBlurUnblurEntrance {
  0%   { opacity: 0; filter: blur(24px); transform: scale(0.85) translateY(20px); }
  65%  { opacity: 1; filter: blur(4px); transform: scale(1.02) translateY(-2px); }
  100% { opacity: 1; filter: blur(0px); transform: scale(1) translateY(0); }
}

/* 2. Hover Backdrop Blur Amplification */
.em-blur-toast:hover {
  transform: scale(1.04) translateY(-4px);
  backdrop-filter: blur(32px);
}

/* 3. HTML Markup Structure */
<div class="em-toast-container toast-pos-top-right">
  <div class="em-blur-toast toast-variant-success toast-show" role="status" aria-live="polite">
    <div class="toast-body-flex">
      <div class="toast-icon-wrapper">...</div>
      <div class="toast-details">
        <h4 class="toast-title">Gallery Published!</h4>
        <p class="toast-message">High-res portfolio live.</p>
      </div>
    </div>
  </div>
</div>
